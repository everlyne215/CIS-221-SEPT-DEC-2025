from flask import Flask, render_template_string import sqlite3 app = Flask(__name__) HTML = """ Fashion Store Inventory

Fashion Store Inventory Overview

{% for item in data %} {% endfor %}
Item Bought Sold In Stock Revenue ($)
{{ item[0] }} {{ item[1] }} {{ item[2] }} {{ item[3] }} {{ "%.2f"|format(item[4]) }}
© 2025 Fashion Store. All rights reserved.
""" @app.route("/") def index(): conn = sqlite3.connect("database.db") cursor = conn.cursor() cursor.execute(""" SELECT item_name, total_stock, sold, (total_stock - sold), (sold * price) FROM inventory """) data = cursor.fetchall() conn.close() return render_template_string(HTML, data=data) if __name__ == "__main__": app.run(debug=True)
